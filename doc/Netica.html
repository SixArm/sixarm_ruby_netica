def finalizer(x)
 return if !defined?(x) or x==nil
 begin
  x.respond_to?('free') and x.free()
 rescue
 end
 begin
  x.respond_to?('finalize') and x.finalize()
 rescue
 end
end

def load_net(filepath)
 streamer = Java::NorsysNetica::Streamer.new(filepath)
 net = Java::NorsysNetica::Net.new(streamer)
 net.compile()
 return net
end
